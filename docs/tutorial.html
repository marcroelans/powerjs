class TodoList extends Component {

  renderTodos() {
    const todos = this.data.todos.map(todo => h('div', null, todo));

    return todos;
  }

  render() {
    return(
      h('div', {class: 'todo'},
        h('input', { keyup: (evt, element) => {
        }}),
        this.renderTodos()
      )
    )
  }
}

class TodoList extends Component {

  renderTodos() {
    const todos = this.data.todos.map(todo => h('div', null, todo));

    return todos;
  }

  addTodo(todo) {
    this.data.todos.push(todo);
  }

  render() {
    return(
      h('div', {class: 'todo'},
        h('input', { keyup: (evt, element) => {
          if (evt.key === 'Enter' && element.value !== '') {
            this.addTodo(element.value);
          }
        }}),
        this.renderTodos()
      )
    )
  }
}

class TodoList extends Component {
