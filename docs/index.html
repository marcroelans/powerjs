Power.js

Powerjs is a small but powerful library for building user interfaces with no depenencies.

Stateless Component

            
class Name  extends Power.Component {

  render () {
    return (
      Power.h('h1', `Hello my name is ${this.data.name}`)
    )
  }

}

Power.render(new Name({name: 'Max'}), mount);

Counter Component

            
class Counter extends Power.Component {

  render () {
    return ([
      Power.h('p', 'Counter: ', this.data.counter.toString()),
      Power.h('button', '+', {click: () => { this.up() } }),
      Power.h('button', '-', {click: () => { this.down() } })
    ])
  }

  up () {
    this.setState(() => this.data.counter += 1)
  }

  down () {
    this.setState(() => this.data.counter -= 1)
  }

}

Power.render(new Counter({counter: 1}), mount)

TodoApp

            
class TodoApp extends Power.Component {

  render () {

    return ([
      Power.h('input', {type: 'text', placeholder: 'Insert task', keyup: (e, element) => {
        if(e.keyCode === 13) {
          this.setState(() => this.data.todos.push(element.value))
        }
      }}),
      this.renderTodos()
    ])

  }

  renderTodos () {

    if(this.data.todos.length === 0) {
      return null;
    } else {
      const todos = this.data.todos.map(todo => Power.h('li', todo) )
      return Power.h('ul', todos)
    }

  }

}

Power.render(new TodoApp({todos: []}), mount)
